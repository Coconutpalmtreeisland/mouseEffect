// 선택자, 마우스 커서 엘리먼트 선택
    const mouseCursor = document.querySelector(".mouse__cursor");

    window.addEventListener("mousemove", e => {                     // 마우스 좌표를 화면에 표시 및 마우스 커서 이동
        gsap.to(".mouse__cursor", {
            duration: 0.5,                                          // 0.5초 동안 애니메이션
            left: e.pageX - mouseCursor.clientWidth/2,              // e.pageX와 e.pageY 값을 조정하여 마우스 커서 엘리먼트를 마우스 위치에 따라 이동
            top: e.pageY - mouseCursor.clientHeight/2               // mouseCursor.clientWidth와 mouseCursor.clientHeight는 마우스 커서 엘리먼트의 너비와 높이를 나타냄, 마우스 커서의 중심 지점을 설정
        });
    });
